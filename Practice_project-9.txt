package queue;
import java.util.*;
public class QueueOperation 
{
	public static void main(String[] args) 
	{
		Queue<Integer> q = new LinkedList<>();
		q.add(10);
		q.add(20);
		q.add(30);
		System.out.println("Queue is : "+q);
		q.remove();
		System.out.println("Queue after removing : "+q);
		
	}
}

OUTPUT:

Queue is : [10, 20, 30]
Queue after removing : [20, 30]
