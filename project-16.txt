<script>
class Deal {
  constructor(name, price) {
    this.name = name;
    this.price = price;
  }
  static total(p,q,r){
    return p.price+ q.price+ r.price;
  }
}
const deal1 = new Deal("Laptop", 40000);
const deal2 = new Deal("Television", 5000);
const deal3 = new Deal("Computer", 25000);
console.log("Vendor Name is : "+deal1.name+"\nPrice : "+deal1.price);
console.log("Vendor Name is : "+deal2.name+"\nPrice : "+deal2.price);
console.log("Vendor Name is : "+deal3.name+"\nPrice : "+deal3.price);

console.log("Total Deals Price : "+ Deal.total(deal1,deal2,deal3));

</script>
